var InfrastructureSet = wire.NewSet(
    ProvideDatabase,
    ProvideTracerProvider,
    ProvideTracer,
)

    wire.Build(
        InfrastructureSet,
        RepositorySet,
        ServiceSet,
        HandlerSet,
        NewApp,
